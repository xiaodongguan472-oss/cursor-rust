```go
// internal/updater/manager.go
type manifest struct {
    Version      string                      `json:"version"`
    ReleaseDate  string                      `json:"release_date"`
    ReleaseNotes string                      `json:"release_notes"`
    Platforms    map[string]manifestPlatform `json:"platforms"`
    Mandatory    bool                        `json:"mandatory"`

type manifestPlatform struct {
    URL      string `json:"url"`
    Size     int64  `json:"size"`
    Checksum string `json:"checksum"`
}
```

### 平台 Key 映射规则

```go
// internal/updater/manager.go - currentPlatformKey()
func currentPlatformKey() (string, error) {
    switch runtime.GOOS {
    case "darwin":
        switch runtime.GOARCH {
        case "arm64":  return "macos-arm64", nil   // Apple Silicon Mac
        case "amd64":  return "macos-amd64", nil   // Intel Mac
        }
    case "windows":
        if runtime.GOARCH == "amd64" {
            return "windows-amd64", nil            // Windows 64位
        }
    case "linux":
        if runtime.GOARCH == "amd64" {
            return "linux-amd64", nil              // Linux 64位
        }
    }
    return "", fmt.Errorf("unsupported platform: %s/%s", runtime.GOOS, runtime.GOARCH)
}
```

## 四、状态机设计

更新管理器是一个有限状态机，状态转换如下：

```
                      ┌─────────┐
                      │  idle   │◀──────── 已是最新版本
                      └────┬────┘
                           │ checkNow()
                           ▼
                      ┌──────────┐
               ┌──────│ checking │──────┐
               │      └──────────┘      │
               │ 发现新版本              │ 无新版本 / 错误
               ▼                        ▼
          ┌───────────┐            ┌─────────┐
          │ available │            │  idle   │
          └─────┬─────┘            │  error  │
                │                  └─────────┘
                │ 用户确认下载 (DownloadAvailable)
                ▼
          ┌──────────────┐
          │ downloading  │──────── 下载失败 ──▶ error
          └──────┬───────┘
                 │ 下载完成 + SHA256 校验通过
                 ▼
          ┌──────────────┐
          │ installing   │──────── 安装失败 ──▶ error
          └──────┬───────┘
                 │ 启动安装脚本成功
                 ▼
          ┌──────────────┐
          │ app.Quit()   │  退出应用，安装脚本接管
          │ os.Exit(0)   │
          └──────────────┘
```

### 状态定义

```go
const (
    StateIdle        State = "idle"         // 空闲，无更新

    StateAvailable   State = "available"    // 发现新版本，等待用户确认

    StateReady       State = "ready"        // 下载完成，等待安装
    StateInstalling  State = "installing"   // 正在安装
```

## 五、完整更新流程

### 5.1 后台自动检查

```go
// internal/updater/manager.go
const checkInterval = 20 * time.Minute

func (m *Manager) Start() {
    m.emitState(StateIdle, nil, "", "", false, "")
    go m.loop()
}

func (m *Manager) loop() {
    m.checkNow(false)                    // 启动时立即检查一次
    ticker := time.NewTicker(checkInterval)
    defer ticker.Stop()
    for {
        select {
        case <-m.ctx.Done():
            return
        case <-ticker.C:
            m.checkNow(false)            // 每 20 分钟自动检查
        }
    }
}
```

### 5.2 检查更新

```go
func (m *Manager) fetchUpdateInfo(ctx context.Context) (*UpdateInfo, error) {
    // 1. 从远程服务器拉取 update.json
    req, _ := http.NewRequestWithContext(ctx, http.MethodGet, buildinfo.UpdateBaseURL+"update.json", nil)
    resp, err := m.client.Do(req)

    // 2. 解析 JSON
    var data manifest
    json.NewDecoder(resp.Body).Decode(&data)

    // 3. 语义化版本比较：远程版本 > 本地版本 才认为有更新
    if compareVersions(data.Version, buildinfo.CurrentVersion()) <= 0 {
        return nil, nil  // 已是最新
    }

    // 4. 根据当前平台查找对应资产
    platformKey, _ := currentPlatformKey()
    asset, ok := data.Platforms[platformKey]
    if !ok {
        return nil, errNoSupportedAsset
    }

    return &UpdateInfo{...}, nil
}
```

### 5.3 下载更新包

```go
func (m *Manager) downloadUpdate(ctx context.Context, info *UpdateInfo) (string, error) {
    // 1. HTTP GET 下载更新包
    req, _ := http.NewRequestWithContext(ctx, http.MethodGet, info.Asset.URL, nil)
    resp, _ := m.client.Do(req)

    // 2. 写入临时文件，同时计算 SHA256 和上报进度
    tempFile, _ := os.CreateTemp("", "starry-cursor-refill-update-*"+archiveSuffix(info.Asset.URL))
    hasher := sha256.New()
    progress := newProgressWriter(func(downloaded int64) {
        m.emitProgress(info, downloaded, total)
    })
    io.Copy(io.MultiWriter(tempFile, hasher, progress), resp.Body)

    // 3. SHA256 校验
    expectedChecksum := strings.TrimPrefix(info.Asset.Checksum, "sha256:")
    actualChecksum := hex.EncodeToString(hasher.Sum(nil))
    if !strings.EqualFold(expectedChecksum, actualChecksum) {
        os.Remove(tempFile.Name())
        return "", fmt.Errorf("checksum mismatch: expected %s got %s", expectedChecksum, actualChecksum)
    }

    return tempFile.Name(), nil
}
```

**关键设计：三路 Writer**

```
HTTP Response Body
       │
       ▼
  io.MultiWriter ──┬──▶ tempFile  (磁盘存储)
                   ├──▶ hasher    (SHA256 计算)
                   └──▶ progress  (进度上报，每120ms emit一次)
```

### 5.4 安装更新

下载完成后自动执行安装，流程：

```go
func (m *Manager) downloadAndInstall(info *UpdateInfo) {
    // 1. 下载
    archivePath, err := m.downloadUpdate(ctx, info)

    // 2. 启动平台安装脚本（外部进程）
    m.spawnInstaller(archivePath)

    // 3. 退出当前应用，让安装脚本接管
    m.app.Quit()
    time.Sleep(3 * time.Second)
    os.Exit(0)
}
```

## 六、平台安装脚本

安装脚本以**外部进程**方式启动，等待主程序退出后执行替换，最后重启新版本。

### 6.1 macOS 安装脚本 (Shell)

```sh
#!/bin/sh
set -eu

PID_TO_WAIT="$1"       # 主程序 PID
ARCHIVE_PATH="$2"      # 下载的 .tar.gz 路径
TARGET_APP="$3"         # 当前 .app 包路径
TMP_DIR="$(mktemp -d)"

cleanup() {
  rm -rf "$TMP_DIR"
  rm -f "$ARCHIVE_PATH"
  rm -f "$0"              # 删除自身脚本
}
trap cleanup EXIT

# 等待主程序退出
while kill -0 "$PID_TO_WAIT" 2>/dev/null; do
  sleep 1
done

# 解压新版本
tar -xzf "$ARCHIVE_PATH" -C "$TMP_DIR"

# 找到 .app 包
EXTRACTED_APP="$(find "$TMP_DIR" -maxdepth 1 -type d -name '*.app' | head -n 1)"

# 替换旧版本并启动
rm -rf "$TARGET_APP"
mv "$EXTRACTED_APP" "$TARGET_APP"
open "$TARGET_APP"
```

**macOS 定位 .app 包路径的方法：**

```go
func resolveMacBundlePath(executablePath string) (string, error) {
    // 可执行文件在 xxx.app/Contents/MacOS/二进制 中
    contentsDir := filepath.Dir(filepath.Dir(executablePath))
    appBundlePath := filepath.Dir(contentsDir)  // 得到 xxx.app
    return appBundlePath, nil
}
```

### 6.2 Windows 安装脚本 (PowerShell)

```powershell
param(
  [int]$PidToWait,          # 主程序 PID
  [string]$ArchivePath,     # 下载的 .zip 路径
  [string]$TargetExecutable # 当前 .exe 路径
)

$ErrorActionPreference = "Stop"
$TargetDir = Split-Path -Parent $TargetExecutable
$TempDir = Join-Path ([System.IO.Path]::GetTempPath()) ("starry-cursor-refill-update-" + [System.Guid]::NewGuid().ToString("N"))

New-Item -ItemType Directory -Path $TempDir | Out-Null

try {
  # 等待主程序退出
  while (Get-Process -Id $PidToWait -ErrorAction SilentlyContinue) {
    Start-Sleep -Seconds 1
  }

  # 解压 zip
  Expand-Archive -LiteralPath $ArchivePath -DestinationPath $TempDir -Force

  # 找到 exe 文件（优先匹配同名 exe）
  $ExtractedExecutables = @(Get-ChildItem -LiteralPath $TempDir -File | Where-Object { $_.Extension -ieq ".exe" })
  $TargetExecutableName = [System.IO.Path]::GetFileName($TargetExecutable)
  $PayloadExecutable = $ExtractedExecutables | Where-Object { $_.Name -ieq $TargetExecutableName } | Select-Object -First 1
  if ($null -eq $PayloadExecutable) {
    $PayloadExecutable = $ExtractedExecutables[0]
  }

  # 复制所有非 exe 文件到目标目录（资源文件等）
  Get-ChildItem -LiteralPath $TempDir -Force | ForEach-Object {
    if ($_.PSIsContainer) {
      Copy-Item -LiteralPath $_.FullName -Destination (Join-Path $TargetDir $_.Name) -Recurse -Force
    } elseif ($_.Extension -ine ".exe") {
      Copy-Item -LiteralPath $_.FullName -Destination (Join-Path $TargetDir $_.Name) -Force
    }
  }

  # 替换 exe 并重启
  Copy-Item -LiteralPath $PayloadExecutable.FullName -Destination $TargetExecutable -Force
  Start-Process -FilePath $TargetExecutable
} finally {
  # 清理临时文件和自身
  Remove-Item -LiteralPath $ArchivePath -Force -ErrorAction SilentlyContinue
  Remove-Item -LiteralPath $TempDir -Recurse -Force -ErrorAction SilentlyContinue
  Remove-Item -LiteralPath $MyInvocation.MyCommand.Path -Force -ErrorAction SilentlyContinue
}
```

**Windows 启动脚本的方式：**

```go
cmd := exec.Command(
    "powershell",
    "-NoProfile",
    "-ExecutionPolicy", "Bypass",   // 绕过策略限制
    "-WindowStyle", "Hidden",       // 隐藏窗口
    "-File", scriptPath,
    "-PidToWait", fmt.Sprintf("%d", os.Getpid()),
    "-ArchivePath", archivePath,
    "-TargetExecutable", executablePath,
)
cmd.Start()  // 启动后不等待，主程序随后退出
```

### 6.3 Linux 安装脚本 (Shell)

```sh
#!/bin/sh
set -eu

PID_TO_WAIT="$1"
ARCHIVE_PATH="$2"
TARGET_EXECUTABLE="$3"
TARGET_DIR="$(dirname "$TARGET_EXECUTABLE")"

  rm -rf "$TMP_DIR"
  rm -f "$ARCHIVE_PATH"
  rm -f "$REPLACEMENT_PATH"

```

## 九、版本比较算法

采用语义化版本号比较 (Semantic Versioning)：

```go
// internal/updater/semver.go
type semanticVersion struct {
    major      int    // 主版本号
    minor      int    // 次版本号
    patch      int    // 修订号
    prerelease string // 预发布标识
}

func compareVersions(a, b string) int {
    // 1. 去掉 "v" 前缀
    // 2. 分离预发布标识 (如 "1.0.0-beta")
    // 3. 按 major → minor → patch 逐级比较
    // 4. 正式版 > 预发布版
    // 返回: 1(a>b), -1(a<b), 0(相等)
}
```

**比较逻辑：**
- `0.0.18` > `0.0.17` → 有更新
- `0.0.17` = `0.0.17` → 无更新
- `1.0.0` > `1.0.0-beta` → 正式版优先于预发布版

### 10.3 并发安全

```go
type Manager struct {
    mu sync.Mutex  // 保护状态读写
    state State
    // ...
}
```

所有状态变更都通过 `m.mu.Lock()` 保护，防止并发检查/下载冲突。

### 10.4 安装脚本自清理

三个平台的安装脚本都在 `cleanup` / `finally` 中清理临时文件和自身：
- 删除下载的安装包
- 删除临时解压目录
- 删除脚本自身

## 十一、构建与发布流程

### 11.1 构建命令

```bash
# macOS 上一键构建所有平台
task build:all

# 分别构建
task build:darwin:arm64      # macOS Apple Silicon
task build:darwin:amd64      # macOS Intel
task build:windows:amd64     # Windows 64位
task build:linux:amd64       # Linux 64位
```

### 11.2 发布流程

```bash
# 1. 准备发布资产（在 macOS 上执行）
task release:prepare:darwin

# 2. 生成 update.json 清单
task release:manifest

# 3. 发布到 GitHub Releases
task release:github
```

`release:manifest` 自动生成包含所有平台 URL、文件大小、SHA256 校验和的 `update.json`。

### 11.3 产物格式

| 平台 | 格式 | 安装方式 |
|------|------|---------|
| macOS | `.tar.gz` (含 .app 包) | 解压替换整个 .app 目录 |
| Windows | `.zip` (含 .exe + 资源) | 解压替换 exe 和资源文件 |
| Linux | `.tar.gz` (单可执行文件) | 解压替换可执行文件 |

## 十二、数据流总结

```
用户点击「检查更新」
  │
  ▼
前端 Call.ByName("bridge.UpdaterService.CheckUpdate")
  │
  ▼
Go bridge.CheckUpdate() → manager.CheckNow(true)
  │
  ▼
HTTP GET update.json → 版本比较 → 平台匹配
  │
  ├── 无更新 → emit("update:state", {state:"idle"}) → 前端弹窗「已是最新」
  │
  └── 有更新 → emit("update:state", {state:"available", version:"x.x.x"})
                  │
                  ▼
              前端弹窗「发现新版本，是否更新？」
                  │
                  ├── 取消 → 不操作
                  │
                  └── 确认 → Call.ByName("bridge.UpdaterService.DownloadUpdate")
                              │
                              ▼
                          manager.DownloadAvailable()
                              │
                              ▼
                          HTTP GET 安装包 → 写入临时文件
                          同时: emit("update:progress", {percentage: xx})
                              │
                              ▼
                          SHA256 校验 → 通过
                              │
                              ▼
                          spawnInstaller() → 启动平台安装脚本
                              │
                              ▼
                          app.Quit() + os.Exit(0) → 主程序退出
                              │
                              ▼
                          安装脚本: 等待 PID 退出 → 解压替换 → 启动新版本
```
