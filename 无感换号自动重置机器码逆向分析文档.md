## 3. 注入流程详解

### 注入函数入口

```python
class CursorInjector:
    def inject(self, pro: bool = False) -> tuple[bool, str]:
```

### 完整注入步骤

```
1. 检测 Cursor 安装路径是否存在
2. (Linux) 确保文件可写（pkexec/sudo 提权）
3. 读取 workbench.desktop.main.js 原始内容
4. 修复 CRLF → LF（Windows 文本模式污染）
5. 清除旧注入残留（strip_markers）
6. 注入点 0：绕过完整性检查通知
7. 注入点 1：暴露 window.store + 创建通知系统
8. 注入点 2：Token 轮询 + fetch 拦截 + DOM 监听 + 自动换号
9. (macOS) 整包副本 → 修改 → 重签 → 原子替换
   (Windows/Linux) 直接写入 + 创建 .bak 备份
10. 写后验证（二进制比对）
11. 禁用 Cursor 自动更新
12. 清除 V8 字节码缓存（防止黑屏）
```

## 5. 本地 HTTP 服务

### 服务端代码 (`seamless_server.py`)

```python
"""本地 Token HTTP 服务 — 供 Cursor 注入 JS 轮询获取最新凭证 + 自动换号"""
from __future__ import annotations

import json
import os
import time
import threading
from http.server import HTTPServer, BaseHTTPRequestHandler

_PORT = 14520
_HOST = "127.0.0.1"

_api_client = None
_auto_switch_lock = threading.Lock()
_last_auto_switch: float = 0.0
_AUTO_SWITCH_COOLDOWN = 10  # 秒


def _state_file_path() -> str:
    home = os.path.expanduser("~")
    d = os.path.join(home, ".wuxian-assistant")
    os.makedirs(d, exist_ok=True)
    return os.path.join(d, "seamless_state.json")


STATE_FILE = _state_file_path()


def read_state() -> dict:
    try:
        with open(STATE_FILE, "r", encoding="utf-8") as f:
            return json.load(f)
    except (FileNotFoundError, json.JSONDecodeError):
        return {
            "config": {"enabled": False},
            "accessToken": "",
            "refreshToken": "",
            "email": "",
            "is_new": False,
            "machineIds": {},
        }


def write_state(state: dict):
    tmp = STATE_FILE + ".tmp"
    with open(tmp, "w", encoding="utf-8") as f:
        json.dump(state, f, ensure_ascii=False)
    os.replace(tmp, STATE_FILE)
```

| POST | `/api/ack-new` | JS 确认已消费新机器码，将 `is_new` 置为 `false` |

### `/api/auto-switch` 处理流程

```python
def _handle_auto_switch() -> dict:
    """处理自动换号请求"""

    # 1. 检查设备状态（是否被封禁/激活码是否过期）
    device_r = _api_client.init_device()
    if device_r.get("success"):
        d = device_r.get("data") or {}
        if d.get("banned"):
            return {"success": False, "expired": True, "message": "设备已被封禁"}
        if not d.get("activated"):
            return {"success": False, "expired": True, "message": "激活码已到期"}

    # 2. 检查换号额度
    quota_r = _api_client.refresh_count()
    if quota_r.get("success"):
        qd = quota_r.get("data") or {}
        if not qd.get("unlimited"):
            remain = int(qd.get("over_count", 0))
            if remain <= 0:
                return {"success": False, "expired": True, "message": "换号额度已用完"}

    # 3. 从远程服务器获取新的 email + token
    r = _api_client.get_credentials()
    data = r.get("data") or {}
    email = data.get("email", "")
    token = data.get("token", "")

    # 4. 调用 seamless_switch 生成新机器码并写入状态
    from core.seamless_switch import seamless_switch
    if not seamless_switch(email, token):
        return {"success": False, "message": "写入状态失败"}

    return {"success": True, "email": email, "message": "自动换号成功"}
```

## 6. 自动换号与机器码重置核心逻辑

### `seamless_switch.py` 完整代码

```python
"""无感换号核心逻辑 — 生成 machineIds + 写状态文件 + 更新磁盘"""
from __future__ import annotations

import hashlib
import json
import os
import sys
import uuid
import threading


def _cursor_data_root() -> str:
    """Cursor 数据目录（跨平台）"""
    if sys.platform == "win32":
        appdata = os.getenv("APPDATA")
        if not appdata:
            raise EnvironmentError("APPDATA 环境变量未设置")
        return os.path.join(appdata, "Cursor")
    if sys.platform == "darwin":
        return os.path.abspath(
            os.path.expanduser("~/Library/Application Support/Cursor")
        )
    if sys.platform.startswith("linux"):
        return os.path.abspath(os.path.expanduser("~/.config/Cursor"))
    raise NotImplementedError(f"不支持的操作系统: {sys.platform}")


def generate_machine_ids() -> dict:
    """生成一组新的机器标识"""
    return {
        "devDeviceId": str(uuid.uuid4()),
        "macMachineId": hashlib.sha512(os.urandom(64)).hexdigest(),
        "machineId": hashlib.sha256(os.urandom(32)).hexdigest(),
        "sqmId": "{" + str(uuid.uuid4()).upper() + "}",
    }


def _update_disk_files(machine_ids: dict):
    """后台更新磁盘上的 machineId 文件和 storage.json（不需要 Cursor 退出）"""
    try:
        root = _cursor_data_root()

        # 1. 写入 machineId 文件
        mid_path = os.path.join(root, "machineId")
        try:
            os.makedirs(root, exist_ok=True)
            with open(mid_path, "w", encoding="utf-8") as f:
                f.write(machine_ids["machineId"])
        except OSError:
            pass

        # 2. 更新 storage.json — 合并 telemetry 字段
        storage_path = os.path.join(root, "User", "globalStorage", "storage.json")
        if os.path.exists(storage_path):
            telemetry_map = {
                "telemetry.devDeviceId": machine_ids["devDeviceId"],
                "telemetry.macMachineId": machine_ids["macMachineId"],
                "telemetry.machineId": machine_ids["machineId"],
                "telemetry.sqmId": machine_ids["sqmId"],
                "storage.serviceMachineId": str(uuid.uuid4()),
            }
            try:
                with open(storage_path, "r+", encoding="utf-8") as f:
                    data = json.load(f)
                    data.update(telemetry_map)
                    f.seek(0)
                    json.dump(data, f, indent=4)
                    f.truncate()
            except (OSError, json.JSONDecodeError):
                pass

    except Exception as e:
        print(f"[SeamlessSwitch] Disk update error: {e}")


def seamless_switch(email: str, token: str, refresh_token: str | None = None) -> bool:
    """
    无感换号主函数：
    1. 生成新机器码
    2. 写入状态文件供 JS 轮询拾取
    3. 后台线程更新磁盘文件
    """
    from core.seamless_server import write_state

    machine_ids = generate_machine_ids()

    state = {
        "config": {"enabled": True},
        "accessToken": token,
        "refreshToken": refresh_token or token,
        "email": email,
        "is_new": True,          # ← 标记为新，JS 会读取 machineIds
        "machineIds": machine_ids,
    }

    try:
        write_state(state)
    except Exception as e:
        print(f"[SeamlessSwitch] Failed to write state: {e}")
        return False

    # 磁盘文件更新放在后台线程，不阻塞主流程
    threading.Thread(
        target=_update_disk_files,
        args=(machine_ids,),
        daemon=True,
    ).start()

    return True
```

### 写入的标识字段

| 字段名 | 生成算法 | 格式示例 |
|--------|----------|----------|
| `telemetry.devDeviceId` | `uuid.uuid4()` | `a1b2c3d4-e5f6-7890-abcd-ef1234567890` |
| `telemetry.macMachineId` | `sha512(urandom(64))` | 128字符十六进制字符串 |
| `telemetry.machineId` | `sha256(urandom(32))` | 64字符十六进制字符串 |
| `telemetry.sqmId` | `"{" + UUID4.upper() + "}"` | `{A1B2C3D4-E5F6-7890-ABCD-EF1234567890}` |
| `storage.serviceMachineId` | `uuid.uuid4()` | `a1b2c3d4-e5f6-7890-abcd-ef1234567890` |

### 双写机制

无感换号采用**双写**保证一致性：

1. **内存层（即时生效）：** 注入 JS 通过 `window.store.set()` 直接写入 Cursor 运行时 StorageService
2. **磁盘层（持久化）：** 后台线程异步写入 `machineId` 文件和 `storage.json`

> 注意：无感换号模式下**不写 state.vscdb**（因为 Cursor 运行时 SQLite 文件被锁定），而是通过内存直接写入。手动重置模式则需要先关 Cursor 再写 SQLite。

## 8. Windows 与 macOS 处理差异

### 8.1 注入写入方式

#### Windows

```python
# 直接写入文件
self._backup()  # 创建 .bak 备份
out_bytes = content.encode("utf-8")
with open(self._js_path, "wb") as f:
    f.write(out_bytes)
```

- 直接 `open(path, "wb")` 写入
- 备份存放在原位：`workbench.desktop.main.js.bak`
- 使用二进制模式防止 CRLF 污染

#### macOS

macOS 有 **App Management TCC** 保护（Sonoma/Sequoia），即使用 root 权限直接写 `/Applications` 内的 .app 也会被 EPERM 拒绝。

采用「**整包副本 → 修改 → 重签 → 原子替换**」策略：

```python
def _mac_replace_file_in_app(src_file: str, target_path: str):
    """
    macOS: 绕过 App Management 保护的写入策略
    """
    app_root = _mac_find_app_root(target_path)  # 如 /Applications/Cursor.app
    rel = os.path.relpath(target_path, app_root)
    tmp_app = app_root + ".wxtmp"

    parts = [
        # 1) 清掉可能残留的旧副本
        f"rm -rf '{tmp_app}'",

        # 2) 整包复制（APFS 下 cp -a 走 clonefile，秒级完成）
        f"cp -a '{app_root}' '{tmp_app}'",

        # 3) 副本解锁：清 unchg flag、chmod +w、清扩展属性与隔离
        f"chflags -R nouchg '{tmp_app}'",
        f"chmod -R u+w '{tmp_app}'",
        f"xattr -cr '{tmp_app}'",

        # 4) 把注入后的文件写入副本内的目标相对路径
        f"cp -f '{src_file}' '{tmp_app}/{rel}'",

        # 5) 移除旧签名
        f"(codesign --remove-signature '{tmp_app}' || true)",

        # 6) Apple Silicon: 由内向外签所有嵌套 Helper Bundle
        f"if [ -d '{frameworks_path}' ]; then "
        f"find '{frameworks_path}' -name '*.app' -type d -print0 "
        f"| xargs -0 -I HELPER codesign --force --timestamp=none --sign - 'HELPER'; fi",

        # 7) 再签外层 bundle
        f"codesign --force --timestamp=none --sign - '{tmp_app}'",

        # 8) 原子替换：rm 旧的 + mv 新的
        f"rm -rf '{app_root}'",
        f"mv '{tmp_app}' '{app_root}'",
    ]
    cmd = " && ".join(parts)
    # 通过 osascript 提权执行（弹出密码框）
    script = f'do shell script "{cmd}" with administrator privileges'
    subprocess.run(["osascript", "-e", script], check=True, capture_output=True, timeout=300)
```

- **备份路径**：`~/.wuxian-assistant/workbench.desktop.main.js.bak`（不放在 .app 内，避免被替换时丢失）
- **提权方式**：`osascript "do shell script ... with administrator privileges"`
- **重签名**：ad-hoc 签名（`--sign -`），Apple Silicon 必须签名否则无法执行

### 9.4 完整调用链时序

```
1. [触发] Cursor 向 cursor.sh 发请求 → 返回 429
2. [JS] fetch 拦截器捕获 → 调用 _doSwitch("HTTP 429")
3. [JS] _doSwitch() → POST http://127.0.0.1:14520/api/auto-switch
4. [服务端] _handle_auto_switch()
   4.1 检查设备状态 (init_device)
   4.2 检查额度 (refresh_count)
   4.3 获取新凭证 (get_credentials) → 返回 {email, token}
   4.4 调用 seamless_switch(email, token)
       4.4.1 generate_machine_ids() → 生成4个新ID
       4.4.2 write_state() → 写入 seamless_state.json (is_new=true)
       4.4.3 启动后台线程 → _update_disk_files()
             → 写 machineId 文件
             → 更新 storage.json
5. [服务端] 返回 {success: true}
6. [JS] 下一秒轮询 GET /api/get-token
   6.1 发现 accessToken 变化 → window.store.set(新token)
   6.2 发现 is_new=true → window.store.set(新机器码×4)
   6.3 POST /api/ack-new → 服务端将 is_new 置 false
7. [结果] Cursor 内存中的机器码和账号已全部替换，服务端视为新设备
```
