## 2. 注入目标与路径检测

### 目标文件

Cursor 基于 Electron，核心渲染进程 JS 入口：

```
resources/app/out/vs/workbench/workbench.desktop.main.js
```

### 路径检测代码（跨平台）

```python
import os
import sys
import platform

_WORKBENCH_REL = os.path.join(
    "app", "out", "vs", "workbench", "workbench.desktop.main.js",
)


def _has_workbench(resources_dir: str) -> bool:
    """检查 resources 目录下是否存在 workbench JS"""
    return os.path.isfile(os.path.join(resources_dir, _WORKBENCH_REL))


def cursor_install_root() -> str | None:
    """
    返回包含 workbench JS 的 resources 目录
    严格模式：只认标准官方安装路径
    """
    system = platform.system()

    # ── macOS ──
    if system == "Darwin":
        p = "/Applications/Cursor.app/Contents/Resources"
        return p if _has_workbench(p) else None

    # ── Windows ──
    if system == "Windows":
        appdata = os.getenv("LOCALAPPDATA") or ""
        if not appdata:
            userprofile = os.getenv("USERPROFILE") or ""
            appdata = os.path.join(userprofile, "AppData", "Local")
        root = os.path.join(appdata, "Programs", "cursor")
        if not os.path.isdir(root):
            return None

        # 直接 resources/
        direct = os.path.join(root, "resources")
        if _has_workbench(direct):
            return direct

        # Squirrel 安装：app-x.y.z/resources/
        try:
            app_dirs = sorted(
                (item for item in os.listdir(root)
                 if item.lower().startswith("app-")
                 and os.path.isdir(os.path.join(root, item))),
                reverse=True,  # 取最新版本
            )
        except OSError:
            app_dirs = []
        for item in app_dirs:
            candidate = os.path.join(root, item, "resources")
            if _has_workbench(candidate):
                return candidate
        return None

    # ── Linux ──
    for prefix in ("/usr", "/usr/local", os.path.expanduser("~/.local")):
        for sub in ("share", "lib"):
            candidate = os.path.join(prefix, sub, "cursor", "resources")
            if _has_workbench(candidate):
                return candidate
    return None


def find_workbench_js() -> str | None:
    """定位 workbench.desktop.main.js 完整路径"""
    res_root = cursor_install_root()
    if not res_root:
        return None
    candidate = os.path.join(res_root, _WORKBENCH_REL)
    return candidate if os.path.isfile(candidate) else None
```

### 各平台完整路径

| 平台 | 路径 |
|------|------|
| **Windows** | `%LOCALAPPDATA%\Programs\cursor\resources\app\out\vs\workbench\workbench.desktop.main.js` |
| **Windows (Squirrel)** | `%LOCALAPPDATA%\Programs\cursor\app-x.y.z\resources\app\out\vs\workbench\workbench.desktop.main.js` |
| **macOS** | `/Applications/Cursor.app/Contents/Resources/app/out/vs/workbench/workbench.desktop.main.js` |
| **Linux** | `/usr/share/cursor/resources/app/out/vs/workbench/workbench.desktop.main.js` 或 `/usr/lib/cursor/resources/...` 或 `~/.local/share/cursor/resources/...` |

## 3. 注入流程代码

```python
import os
import re
import shutil
import platform
import subprocess
import tempfile


class CursorInjector:

    def __init__(self):
        self._js_path = find_workbench_js()

    def is_available(self) -> bool:
        return self._js_path is not None and os.path.isfile(self._js_path)

    def is_injected(self) -> bool:
        """检测是否已注入（通过标记）"""
        if not self.is_available():
            return False
        with open(self._js_path, "rb") as f:
            raw = f.read()
        return (b"/*i0*/" in raw and b"/*i1s*/" in raw and b"/*i2s*/" in raw)

    def inject(self) -> tuple[bool, str]:
        """
        注入 JS 代码到 workbench.desktop.main.js
        返回 (success, message)
        """
        if not self.is_available():
            return False, "未检测到 Cursor 安装"

        # 1. 读取原始文件
        with open(self._js_path, "rb") as f:
            raw = f.read()
        content = raw.decode("utf-8")

        # 2. 修复可能的 CRLF 污染
        if "\r\n" in content:
            content = content.replace("\r\n", "\n")

        # 3. 清除旧注入残留
        content = self._strip_markers(content)

        # 4. 注入点 0：绕过完整性检查
        INJECT0_SEARCH = '_showNotification(){'
        INJECT0_REPLACE = '_showNotification(){return;/*i0*/'
        if INJECT0_SEARCH not in content:
            return False, "未找到注入点 0，Cursor 版本可能不兼容"
        content = content.replace(INJECT0_SEARCH, INJECT0_REPLACE, 1)

        # 5. 注入点 1：暴露 window.store
        INJECT1_SEARCH = 'this.database.getItems()))'
        if INJECT1_SEARCH not in content:
            return False, "未找到注入点 1，Cursor 版本可能不兼容"
        content = content.replace(INJECT1_SEARCH, INJECT1_SEARCH + INJECT1_CODE, 1)

        # 6. 注入点 2：Token 轮询 + 自动换号（紧跟 i1e 后面）
        i1e_pos = content.find("/*i1e*/")
        if i1e_pos < 0:
            return False, "注入点 1 插入异常"
        insert_pos = i1e_pos + len("/*i1e*/")
        content = content[:insert_pos] + INJECT2_CODE + content[insert_pos:]

        # 7. 写入文件（平台差异处理见第9节）
        self._platform_write(content)

        # 8. 写后验证
        with open(self._js_path, "rb") as f:
            verify = f.read()
        if verify != content.encode("utf-8"):
            return False, "写入验证失败"

        # 9. 清除 V8 字节码缓存（不清会导致 Electron 加载旧编译缓存 → 黑屏）
        self._clear_v8_cache()

        return True, "注入成功，重启 Cursor 后生效"
```

## 4. 注入点 0：绕过完整性检查

### 作用

Cursor 会校验自身 JS 文件完整性，如果发现被修改会弹通知。此注入让通知函数直接 return。

### 代码

**搜索：**
```javascript
_showNotification(){
```

**替换为：**
```javascript
_showNotification(){return;/*i0*/
```

### 原理

Cursor 的完整性检查逻辑会调用 `_showNotification()` 弹出警告，注入后该函数被短路，直接 `return` 不执行任何操作。

### 关键原理

注入位置 `this.database.getItems()))` 处于 Cursor 的 `StorageService` 初始化流程中。此处的 `this` 就是 `StorageService` 实例，具备：

- `this.get(key, scope)` — 读取存储值
- `this.set(key, value, scope)` — 写入存储值

其中 `scope = -1` 代表全局作用域。绑定后，后续代码可通过 `window.store` 直接操控 Cursor 内部的所有配置项，包括 telemetry ID。

## 8. 机器码生成与磁盘写入

### `seamless_switch.py` 完整代码

```python
import hashlib
import json
import os
import sys
import uuid
import threading


def _cursor_data_root() -> str:
    """Cursor 用户数据目录"""
    if sys.platform == "win32":
        appdata = os.getenv("APPDATA")
        if not appdata:
            raise EnvironmentError("APPDATA 环境变量未设置")
        return os.path.join(appdata, "Cursor")
    if sys.platform == "darwin":
        return os.path.expanduser("~/Library/Application Support/Cursor")
    if sys.platform.startswith("linux"):
        return os.path.expanduser("~/.config/Cursor")
    raise NotImplementedError(f"不支持的操作系统: {sys.platform}")


def generate_machine_ids() -> dict:
    """生成一组全新的随机机器标识"""
    return {
        "devDeviceId":   str(uuid.uuid4()),                               # UUID v4
        "macMachineId":  hashlib.sha512(os.urandom(64)).hexdigest(),      # 128字符 hex
        "machineId":     hashlib.sha256(os.urandom(32)).hexdigest(),      # 64字符 hex
        "sqmId":         "{" + str(uuid.uuid4()).upper() + "}",           # {大写UUID}
    }


def _update_disk_files(machine_ids: dict):
    """
    后台线程：更新磁盘上的 machineId 文件和 storage.json
    不需要关闭 Cursor 即可写入
    """
    try:
        root = _cursor_data_root()

        # ── 文件1：machineId ──
        # 纯文本文件，内容为 machineId 值
        mid_path = os.path.join(root, "machineId")
        try:
            os.makedirs(root, exist_ok=True)
            with open(mid_path, "w", encoding="utf-8") as f:
                f.write(machine_ids["machineId"])
            print(f"[Disk] machineId 已更新: {mid_path}")
        except OSError as e:
            print(f"[Disk] machineId 写入失败: {e}")

        # ── 文件2：storage.json ──
        # 合并 telemetry 字段到已有配置中
        storage_path = os.path.join(root, "User", "globalStorage", "storage.json")
        if os.path.exists(storage_path):
            telemetry_map = {
                "telemetry.devDeviceId":     machine_ids["devDeviceId"],
                "telemetry.macMachineId":    machine_ids["macMachineId"],
                "telemetry.machineId":       machine_ids["machineId"],
                "telemetry.sqmId":           machine_ids["sqmId"],
                "storage.serviceMachineId":  str(uuid.uuid4()),  # 额外生成
            }
            try:
                with open(storage_path, "r+", encoding="utf-8") as f:
                    data = json.load(f)
                    data.update(telemetry_map)
                    f.seek(0)
                    json.dump(data, f, indent=4)
                    f.truncate()
                print(f"[Disk] storage.json 已更新: {storage_path}")
            except (OSError, json.JSONDecodeError) as e:
                print(f"[Disk] storage.json 更新失败: {e}")

    except Exception as e:
        print(f"[Disk] 更新异常: {e}")


def seamless_switch(email: str, token: str, refresh_token: str | None = None) -> bool:
    """
    无感换号主函数：
      1. 生成新机器码
      2. 写入 seamless_state.json（供 JS 轮询拾取）
      3. 后台线程更新磁盘文件（保持持久化一致性）
    """
    from seamless_server import write_state

    # 1. 生成新机器标识
    machine_ids = generate_machine_ids()

    # 2. 写入状态文件
    state = {
        "config": {"enabled": True},
        "accessToken": token,
        "refreshToken": refresh_token or token,
        "email": email,
        "is_new": True,              # ← JS 看到此标志会写入机器码
        "machineIds": machine_ids,   # ← 新机器码，JS 会用 window.store.set 写入
    }

    try:
        write_state(state)
    except Exception as e:
        print(f"[SeamlessSwitch] 写状态文件失败: {e}")
        return False

    # 3. 后台线程更新磁盘（不阻塞主流程）
    threading.Thread(
        target=_update_disk_files,
        args=(machine_ids,),
        daemon=True,
    ).start()

    print(f"[SeamlessSwitch] 已排队: {email}")
    return True
```

## 9. Win / Mac / Linux 注入写入差异

### Windows — 直接写入

```python
def inject_windows(js_path: str, content: str):
    """Windows: 备份 + 直接二进制写入"""
    # 备份
    bak = js_path + ".bak"
    if not os.path.exists(bak):
        shutil.copy2(js_path, bak)

    # 二进制模式写入（防止 Python 文本模式自动转 CRLF）
    out_bytes = content.encode("utf-8")
    with open(js_path, "wb") as f:
        f.write(out_bytes)
```

### macOS — 整包副本 + 重签 + 原子替换

macOS Sonoma/Sequoia 有 **App Management TCC** 保护，直接写 `.app` 内文件即使 root 也会 EPERM。

```python
def inject_macos(js_path: str, content: str):
    """macOS: 副本 → 修改 → 重签 → 原子替换"""
    # 备份到用户目录（不放 .app 内）
    bak_dir = os.path.expanduser("~/.wuxian-assistant")
    os.makedirs(bak_dir, exist_ok=True)
    bak = os.path.join(bak_dir, "workbench.desktop.main.js.bak")
    if not os.path.exists(bak):
        shutil.copy2(js_path, bak)

    # 写入临时文件
    fd, tmp = tempfile.mkstemp(suffix=".js")
    with os.fdopen(fd, "wb") as f:
        f.write(content.encode("utf-8"))

    try:
        # 定位 .app 根目录（如 /Applications/Cursor.app）
        app_root = js_path
        while app_root and app_root != "/":
            if app_root.endswith(".app"):
                break
            app_root = os.path.dirname(app_root)

        rel = os.path.relpath(js_path, app_root)
        tmp_app = app_root + ".wxtmp"
        frameworks = os.path.join(tmp_app, "Contents", "Frameworks")

        cmd = " && ".join([
            f"rm -rf '{tmp_app}'",                       # 清旧副本
            f"cp -a '{app_root}' '{tmp_app}'",           # APFS clonefile 整包复制
            f"chflags -R nouchg '{tmp_app}'",            # 清 unchg flag
            f"chmod -R u+w '{tmp_app}'",                 # 加写权限
            f"xattr -cr '{tmp_app}'",                    # 清扩展属性/隔离
            f"cp -f '{tmp}' '{tmp_app}/{rel}'",          # 写入修改后的文件

            # 移除旧签名
            f"(codesign --remove-signature '{tmp_app}' || true)",

            # Apple Silicon：必须由内向外签所有嵌套 Helper Bundle
            f"if [ -d '{frameworks}' ]; then "
            f"find '{frameworks}' -name '*.app' -type d -print0 "
            f"| xargs -0 -I HELPER "
            f"codesign --force --timestamp=none --sign - 'HELPER'; fi",

            # 签外层 bundle
            f"codesign --force --timestamp=none --sign - '{tmp_app}'",

            # 原子替换
            f"rm -rf '{app_root}'",
            f"mv '{tmp_app}' '{app_root}'",
        ])

        # 通过 osascript 弹密码框提权
        script = f'do shell script "{cmd}" with administrator privileges'
        subprocess.run(
            ["osascript", "-e", script],
            check=True, capture_output=True, timeout=300
        )
    finally:
        try: os.unlink(tmp)
        except: pass
```

**macOS 关键细节：**
- `cp -a` 在 APFS 上走 clonefile，秒级完成整包复制
- 必须清 `unchg` 标志、扩展属性、隔离属性
- Apple Silicon 必须 `codesign` ad-hoc 签名（`--sign -`），否则无法启动
- `--deep` 在 macOS 15 不可靠，必须逐个显式签嵌套的 Helper .app
- 提权方式：`osascript "do shell script ... with administrator privileges"`

### Linux — pkexec/sudo 提权

```python
def inject_linux(js_path: str, content: str):
    """Linux: 提权写入"""
    parent = os.path.dirname(js_path)

    # 检查并获取写权限
    if not os.access(parent, os.W_OK) or not os.access(js_path, os.W_OK):
        uid = os.getuid()
        gid = os.getgid()
        cmd = f"chown -R {uid}:{gid} '{parent}'"
        for elevator in ["pkexec", "sudo"]:
            if shutil.which(elevator):
                try:
                    subprocess.run(
                        [elevator, "sh", "-c", cmd],
                        check=True, capture_output=True, timeout=60
                    )
                    break
                except subprocess.CalledProcessError:
                    continue

    # 备份 + 写入
    bak = js_path + ".bak"
    if not os.path.exists(bak):
        shutil.copy2(js_path, bak)

    with open(js_path, "wb") as f:
        f.write(content.encode("utf-8"))
```

### 注入后必须执行：清除 V8 缓存

```python
def clear_v8_cache():
    """
    清除 Cursor 的 V8 字节码缓存
    Electron 会编译 JS 为字节码缓存，修改 JS 后如果不清缓存，
    Electron 仍会加载旧的字节码 → 注入代码不生效甚至黑屏
    """
    roots = [_cursor_data_root()]

    if sys.platform == "win32":
        local = os.getenv("LOCALAPPDATA")
        if local:
            p = os.path.join(local, "Cursor")
            if os.path.isdir(p):
                roots.append(p)

    for root in roots:
        if not os.path.exists(root):
            continue
        for item in os.listdir(root):
            full = os.path.join(root, item)
            if os.path.isdir(full) and "Cache" in item:
                try:
                    shutil.rmtree(full)
                except Exception:
                    pass
```

| 操作 | Windows | macOS | Linux |

## 10. 触发条件与频率控制

### 三种自动触发方式

| # | 触发方式 | 检测机制 | 具体条件 |
|---|----------|----------|----------|
| 1 | **HTTP 拦截** | 劫持 `window.fetch` | Cursor → `cursor.sh`/`cursor.com`/`cursor.so` 的请求返回 **401/403/429** |
| 2 | **DOM 监听** | `MutationObserver` | 页面新增节点包含 "usage limit"、"free trial ended" 等关键词 |
| 3 | **未登录检测** | 每秒 `window.store.get()` | `cursorAuth/accessToken` 为空 |

### DOM 匹配正则

```javascript
/you.{0,5}(have|ve).{0,5}(hit|reached).{0,10}(usage|limit)|too many free.{0,5}trial|free trial.{0,10}(ended|expired|over)/i
```

匹配例子：
- `"You have hit the usage limit"`
- `"You've reached your usage limit"`
- `"Too many free trial accounts"`
- `"Your free trial has ended"`
- `"Free trial expired"`

### 频率控制参数

| 参数 | 值 | 说明 |
|------|-----|------|
| 轮询间隔 | 1 秒 | `setInterval(..., 1000)` |
| 基础冷却 | 30 秒 | 成功/失败后最短等待时间 |
| 最大冷却 | 5 分钟 | 连续失败后最长等待时间 |
| 退避策略 | 2^n 指数 | 30s → 60s → 120s → 240s → 300s |
| DOM 暂停 | 60 秒 | DOM 触发后暂停监听 |
| 未登录检查 | 5 秒间隔 | `_lastNoLoginCheck` |
| 请求超时 | 轮询 3s / 换号 15s | `AbortSignal.timeout()` |
